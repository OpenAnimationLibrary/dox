750 /*
751 parsing preset string with QString::split().
752 !NOTE! fx/fy (camera size in inch) and xoffset/yoffset (camera offset used in
753 cleanup camera) are optional,
754 in order to keep compatibility with default (Harlequin's) reslist.txt
755 */
756
757 QStringList tokens = str.split(",", QString::SkipEmptyParts);
758
759 if (!(tokens.count() == 3 ||
760 (!forCleanup && tokens.count() == 4) || /*- with "fx x fy" token -*/
761 (forCleanup &&
762 tokens.count() ==
763 6))) /*- with "fx x fy", xoffset and yoffset tokens -*/
764 return false;
765 /*- name -*/
766 name = tokens[0];
767
768 /*- xres, yres (like: 1024x768) -*/
769 QStringList values = tokens[1].split("x");
770 if (values.count() != 2) return false;
771 bool ok;
772 xres = values[0].toInt(&ok);
773 if (!ok) return false;
774 yres = values[1].toInt(&ok);
775 if (!ok) return false;
776
777 if (tokens.count() >= 4) {
778 /*- fx, fy -*/
779 values = tokens[2].split("x");
780 if (values.count() != 2) return false;
781 fx = values[0].toDouble(&ok);
782 if (!ok) return false;
783 fy = values[1].toDouble(&ok);
784 if (!ok) return false;
785
786 /*- xoffset, yoffset -*/
787 if (forCleanup) {
788 xoffset = tokens[3];
789 yoffset = tokens[4];
790 /*- remove single space -*/
791 if (xoffset.startsWith(' ')) xoffset.remove(0, 1);
792 if (yoffset.startsWith(' ')) yoffset.remove(0, 1);
793 }
794 }
795

797 ar = aspectRatioStringToValue(tokens.last());
