152 // in case the handle length is 0 on one side, take the oppositte handle to
153 // calculate the angle
154 auto handleAngleLength = [](TPointParamP handle, TPointParamP cp,
155 TPointParamP opposite, double f,
156 bool isLeft = true) {
157 TPointD vec_h_cp = handle->getValue(f) - cp->getValue(f);
158 double angle;
159 if (vec_h_cp.x == 0 && vec_h_cp.y == 0) {
160 TPointD vec_h_op = handle->getValue(f) - opposite->getValue(f);
161 angle = std::atan2(vec_h_op.y, vec_h_op.x);
162 } else
163 angle = std::atan2(vec_h_cp.y, vec_h_cp.x);
164
165 // make the angle continuous
166 if (isLeft && angle < 0) angle += M_2PI;
167 double length =
168 std::sqrt(vec_h_cp.x * vec_h_cp.x + vec_h_cp.y * vec_h_cp.y);
169 return TPointD(angle, length);
170 };
171 auto angleLengthToPos = [](TPointD anLen) {
172 return TPointD(anLen.y * std::cos(anLen.x), anLen.y * std::sin(anLen.x));
173 };

178 std::set<double>::iterator nextIt = frames.upper_bound(frame);
179 bool isNotInSegment = getCurrentParamSet()->isKeyframe(frame) ||
180 prevIt == frames.begin() || nextIt == frames.end();
181 if (prevIt != frames.begin()) prevIt--;
182
183 int i;
184 QList<TPointD> points;
185 int pointCount = getCurrentParamSet()->getParamCount();

187 // control point case or the current frame is not between the keys

190 points.push_back(pointParam->getValue(frame));
191 } else {

195 if (i % 3 == 2) { // left handle
196 TPointParamP left_Param = getCurrentParamSet()->getParam(i);
197 TPointParamP cp_Param = getCurrentParamSet()->getParam(i + 1);
198 TPointParamP right_Param = (i == pointCount - 2)

201

206 // linear interpolation of angle & length
