450 // Here is the subdivision strategy:
451 // - First, since cache tiles are newly ALLOCATED, we impose the raster
452 // size restriction on them directly
453 // - Then, we check that the memory requirement for the fx (max raster size
454 // that it will allocate) is little.
455 // - As an exception to previous point, if the memory requirement stagnates
456 // near the same memory size, quit
457 // NOTE: Level images pass here, but haven't any fx. So they are currently not
458 // subdivided.
459
460 // Retrieve the memory requirement for this input.

462
463 // In case memReq < 0, we assume a strong subdivision denial, just as if the
464 // usage was
465 // explicitly set as unsubdividable.
466 if (memReq < 0) {
467 results.push_back(rect);
468 return;
469 }
470
471 if ((memReq > info.m_maxTileSize && memReq < dropTol) ||
472 TRasterFx::memorySize(rect, info.m_bpp) > info.m_maxTileSize) {
473 std::vector<TRectD> subTileRects;
474 subdivideIntoSmallerTiles(rect, subTileRects);
475

477 TRectD subTileRect(subTileRects.back());

479

482 memReq - (memReq >> 2));
483
484 // The newly required memory must be under 3/4 of the previous.
485 // This is required in order to make it worth subdividing.
486 }

489 }
